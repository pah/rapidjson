64 //! Result of parsing (wraps ParseErrorCode)

74  \endcode

79  //! Default constructor, no error.
80  ParseResult() : code_(kParseErrorNone), offset_(0) {}
81  //! Constructor to set an error.
82  ParseResult(ParseErrorCode code, size_t offset) : code_(code), offset_(offset) {}
83 
84  //! Get the error code.
85  ParseErrorCode Code() const { return code_; }
86  //! Get the error offset, if \ref IsError(), 0 otherwise.
87  size_t Offset() const { return offset_; }
88 
89  //! Conversion to \c bool, returns \c true, iff !\ref IsError().
90  operator bool() const { return !IsError(); }
91  //! Whether the result is an error.
92  bool IsError() const { return code_ != kParseErrorNone; }
93 
94  bool operator==(const ParseResult& that) const { return code_ == that.code_; }
95  bool operator==(ParseErrorCode code) const { return code_ == code; }
96  friend bool operator==(ParseErrorCode code, const ParseResult & err) { return code == err.code_; }
97 
98  //! Reset error code.
99  void Clear() { Set(kParseErrorNone); }
100  //! Update error code and offset.
101  void Set(ParseErrorCode code, size_t offset = 0) { code_ = code; offset_ = offset; }
102 
103 private:
104  ParseErrorCode code_;
