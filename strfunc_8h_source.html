27 //! Custom strlen() which works on different character types.
28 /*! \tparam Ch Character type (e.g. char, wchar_t, short)
29  \param s Null-terminated input string.
30  \return Number of characters in the string.
31  \note This has the same semantics as strlen(), the return value is not number of Unicode codepoints.
32 */
33 template <typename Ch>
34 inline SizeType StrLen(const Ch* s) {
35  const Ch* p = s;
36  while (*p) ++p;
37  return SizeType(p - s);
38 }
