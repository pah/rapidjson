138  void PrettyPrefix(Type type) {
139  (void)type;
140  if (Base::level_stack_.GetSize() != 0) { // this value is not at root
141  typename Base::Level* level = Base::level_stack_.template Top<typename Base::Level>();
142 
143  if (level->inArray) {
144  if (level->valueCount > 0) {
145  Base::os_.Put(','); // add comma if it is not the first element in array
146  Base::os_.Put('\n');
147  }
148  else
149  Base::os_.Put('\n');
150  WriteIndent();
151  }
152  else { // in object
153  if (level->valueCount > 0) {
154  if (level->valueCount % 2 == 0) {
155  Base::os_.Put(',');
156  Base::os_.Put('\n');
157  }
158  else {
159  Base::os_.Put(':');
160  Base::os_.Put(' ');
161  }
162  }
163  else
164  Base::os_.Put('\n');
165 
166  if (level->valueCount % 2 == 0)
167  WriteIndent();
168  }
169  if (!level->inArray && level->valueCount % 2 == 0)
170  RAPIDJSON_ASSERT(type == kStringType); // if it's in object, then even number should be a name
171  level->valueCount++;
172  }
173  else
174 #ifdef RAPIDJSON_ACCEPT_ANY_ROOT
175  if (!Base::acceptAnyRoot_)
176 #endif
177  RAPIDJSON_ASSERT(type == kObjectType || type == kArrayType);
178  }
